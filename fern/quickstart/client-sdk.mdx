## Installation and Setup

<Tabs>
  <Tab title="Web SDK">
    Build browser-based voice interfaces:

    ```bash
    npm install @vapi-ai/web
    ```

    ```typescript
    import Vapi from '@vapi-ai/web';

    const vapi = new Vapi('YOUR_PUBLIC_API_KEY');

    // Start voice conversation
    vapi.start('YOUR_ASSISTANT_ID');

    // Listen for events
    vapi.on('call-start', () => console.log('Call started'));
    vapi.on('call-end', () => console.log('Call ended'));
    vapi.on('message', (message) => {
      if (message.type === 'transcript') {
        console.log(`${message.role}: ${message.transcript}`);
      }
    });
    ```
  </Tab>

  <Tab title="Python Client">
    Use Python for client-side voice applications:

    ```bash
    pip install vapi-python-client
    ```

    ```python
    from vapi_client import VapiClient

    # Initialize client
    vapi = VapiClient(api_key="YOUR_PUBLIC_API_KEY")

    # Start voice session
    session = vapi.start_session(assistant_id="YOUR_ASSISTANT_ID")

    # Handle events
    @vapi.on('transcript')
    def on_transcript(message):
        print(f"{message.role}: {message.text}")

    # Start listening
    vapi.listen()
    ```
  </Tab>

  <Tab title="React Native">

    ```jsx
    export default () => (
      <VapiProvider apiKey="YOUR_PUBLIC_API_KEY">
        <VoiceApp />
      </VapiProvider>
    );
    ```

    Build native iOS voice apps:


## Voice Widget Implementation

<Tabs>
  <Tab title="Web SDK">
    Create a voice widget for your website:

    ```typescript
    import Vapi from '@vapi-ai/web';

    function createVoiceWidget(config: { apiKey: string; assistantId: string }) {
      const vapi = new Vapi(config.apiKey);
      let isConnected = false;

      // Create widget UI
      const widget = document.createElement('div');
      widget.style.cssText = `
        position: fixed;
        bottom: 24px;
        right: 24px;
        z-index: 1000;
      `;

      const button = document.createElement('button');
      button.textContent = 'ðŸŽ¤ Talk';
      button.style.cssText = `
        background: #12A594;
        color: white;
        border: none;
        border-radius: 50px;
        padding: 16px 24px;
        cursor: pointer;
      `;

      widget.appendChild(button);
      document.body.appendChild(widget);

      // Event handlers
      button.addEventListener('click', () => {
        if (isConnected) {
          vapi.stop();
        } else {
          vapi.start(config.assistantId);
        }
      });

      vapi.on('call-start', () => {
        isConnected = true;
        button.textContent = 'ðŸ”´ End Call';
        button.style.background = '#ff4444';
      });

      vapi.on('call-end', () => {
        isConnected = false;
        button.textContent = 'ðŸŽ¤ Talk';
        button.style.background = '#12A594';
      });

      return { start: () => vapi.start(config.assistantId), stop: () => vapi.stop() };
    }

    // Usage
    const voiceWidget = createVoiceWidget({
      apiKey: 'YOUR_PUBLIC_API_KEY',
      assistantId: 'YOUR_ASSISTANT_ID'
    });
    ```

  <Tab title="React Native">
    Build a voice widget component:

    ```jsx

    import { View, TouchableOpacity, Text, StyleSheet } from 'react-native';
    import { useVapi } from '@vapi-ai/react-native';

    const VoiceWidget = ({ assistantId }) => {
      const { start, stop, isConnected, transcript } = useVapi();

      return (
        <View style={styles.widget}>
          <TouchableOpacity
            style={[styles.button, isConnected && styles.connectedButton]}
            onPress={() => isConnected ? stop() : start(assistantId)}
          >
            <Text style={styles.buttonText}>
              {isConnected ? 'ðŸ”´ End Call' : 'ðŸŽ¤ Start Call'}
            </Text>
          </TouchableOpacity>
          
          {transcript.length > 0 && (
            <View style={styles.transcript}>
              {transcript.map((msg, i) => (
                <Text key={i} style={styles.message}>
                  {msg.role}: {msg.text}
                </Text>
              ))}

      );

      widget: {
        position: 'absolute',
        bottom: 24,
        right: 24,

        padding: 12,
        marginTop: 8,
        maxWidth: 300,
      },
    ```

  <Tab title="Flutter">
    Create a Flutter voice widget:

    ```dart
    import 'package:flutter/material.dart';
    import 'package:vapi_flutter/vapi_flutter.dart';

    class VoiceWidget extends StatefulWidget {
      final String assistantId;
      
      const VoiceWidget({Key? key, required this.assistantId}) : super(key: key);

      @override
      _VoiceWidgetState createState() => _VoiceWidgetState();
    }

    class _VoiceWidgetState extends State<VoiceWidget> {
      final VapiClient _vapi = VapiClient('YOUR_PUBLIC_API_KEY');
      bool _isConnected = false;
      List<TranscriptMessage> _transcript = [];

      @override
      void initState() {
        super.initState();
        
        _vapi.onCallStart = () {
          setState(() => _isConnected = true);
        };
        
        _vapi.onCallEnd = () {
          setState(() => _isConnected = false);
        };
        
        _vapi.onTranscript = (message) {
          setState(() => _transcript.add(message));
        };
      }

      @override
      Widget build(BuildContext context) {
        return Positioned(
          bottom: 24,
          right: 24,
          child: Column(
            crossAxisAlignment: CrossAxisAlignment.end,
            children: [
              if (_transcript.isNotEmpty)
                Container(
                  constraints: BoxConstraints(maxWidth: 300, maxHeight: 200),
                  decoration: BoxDecoration(
                    color: Colors.white,
                    borderRadius: BorderRadius.circular(12),
                    boxShadow: [BoxShadow(color: Colors.black26, blurRadius: 8)],
                  ),
                  child: ListView.builder(
                    itemCount: _transcript.length,
                    itemBuilder: (context, index) {
                      final msg = _transcript[index];
                      return Padding(
                        padding: EdgeInsets.all(8),
                        child: Text('${msg.role}: ${msg.text}'),
                      );
                    },
                  ),
                ),
              SizedBox(height: 8),
              FloatingActionButton.extended(
                onPressed: () {
                  if (_isConnected) {
                    _vapi.stop();
                  } else {
                    _vapi.start(widget.assistantId);
                  }
                },
                backgroundColor: _isConnected ? Colors.red : Color(0xFF12A594),
                label: Text(_isConnected ? 'ðŸ”´ End Call' : 'ðŸŽ¤ Start Call'),
              ),
            ],
          ),
        );
      }
    }
    ```

  <Tab title="iOS">
    Implement a voice widget in iOS:

    ```swift
    import UIKit
    import VapiSDK

    class VoiceWidget: UIView {
        private let vapi: VapiClient
        private var isConnected = false
        private let assistantId: String
        
        private lazy var button: UIButton = {
            let btn = UIButton(type: .system)
            btn.setTitle("ðŸŽ¤ Start Call", for: .normal)
            btn.backgroundColor = UIColor(red: 0.07, green: 0.65, blue: 0.58, alpha: 1.0)
            btn.setTitleColor(.white, for: .normal)
            btn.layer.cornerRadius = 25
            btn.titleLabel?.font = .boldSystemFont(ofSize: 16)
            btn.addTarget(self, action: #selector(buttonTapped), for: .touchUpInside)
            return btn
        }()
        
        init(apiKey: String, assistantId: String) {
            self.vapi = VapiClient(apiKey: apiKey)
            self.assistantId = assistantId
            super.init(frame: .zero)
            
            setupUI()
            vapi.delegate = self
        }
        
        required init?(coder: NSCoder) {
            fatalError("init(coder:) has not been implemented")
        }
        
        private func setupUI() {
            addSubview(button)
            button.translatesAutoresizingMaskIntoConstraints = false
            NSLayoutConstraint.activate([
                button.widthAnchor.constraint(equalToConstant: 140),
                button.heightAnchor.constraint(equalToConstant: 50),
                button.centerXAnchor.constraint(equalTo: centerXAnchor),
                button.centerYAnchor.constraint(equalTo: centerYAnchor)
            ])
        }
        
        @objc private func buttonTapped() {
            if isConnected {
                vapi.stop()
            } else {
                vapi.start(assistantId: assistantId)
            }
        }
    }

    extension VoiceWidget: VapiClientDelegate {
        func vapiCallDidStart() {
            isConnected = true
            button.setTitle("ðŸ”´ End Call", for: .normal)
            button.backgroundColor = .systemRed
        }
        
        func vapiCallDidEnd() {
            isConnected = false
            button.setTitle("ðŸŽ¤ Start Call", for: .normal)
            button.backgroundColor = UIColor(red: 0.07, green: 0.65, blue: 0.58, alpha: 1.0)
        }
    }
    ```
